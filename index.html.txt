        // 캔버스 자유 필기 연습장 초기화 (아이패드, 터치 기기 100% 호환 안전 연산)
        function initDrawingCanvas() {
            state.canvas = document.getElementById('scratch-canvas');
            if (!state.canvas) return;
            state.ctx = state.canvas.getContext('2d');

            const parent = state.canvas.parentNode;
            state.canvas.width = parent.clientWidth;
            state.canvas.height = 200;

            state.ctx.strokeStyle = '#4f46e5';
            state.ctx.lineWidth = 3;
            state.ctx.lineCap = 'round';

            const getTouchPos = (e) => {
                const rect = state.canvas.getBoundingClientRect();
                const clientX = e.touches ? e.touches[0].clientX : e.clientX;
                const clientY = e.touches ? e.touches[0].clientY : e.clientY;
                return {
                    x: clientX - rect.left,
                    y: clientY - rect.top
                };
            };

            const onStart = (e) => {
                state.isDrawing = true;
                const pos = getTouchPos(e);
                state.ctx.beginPath();
                state.ctx.moveTo(pos.x, pos.y);
            };

            const onMove = (e) => {
                if (!state.isDrawing) return;
                e.preventDefault(); // 스크롤 차단
                const pos = getTouchPos(e);
                state.ctx.lineTo(pos.x, pos.y);
                state.ctx.stroke();
            };

            const onStop = () => {
                state.isDrawing = false;
            };

            state.canvas.addEventListener('mousedown', onStart);
            state.canvas.addEventListener('mousemove', onMove);
            window.addEventListener('mouseup', onStop);

            state.canvas.addEventListener('touchstart', onStart, { passive: false });
            state.canvas.addEventListener('touchmove', onMove, { passive: false });
            window.addEventListener('touchend', onStop);
        }

        function clearCanvas() {
            if (state.ctx && state.canvas) {
                state.ctx.clearRect(0, 0, state.canvas.width, state.canvas.height);
            }
        }
